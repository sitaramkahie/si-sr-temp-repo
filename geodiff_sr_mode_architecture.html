        STYLE["Layer-wise FiLM styles<br/>s0, s1, s2, s3<br/>Each B x Ds"]

        STYLE --> D256
        STYLE --> D512
        STEM --> D128
        DOWN1 --> D64

    subgraph COMPOSE["SR Composition and Physical Consistency"]
        ADD["Residual addition<br/>x_pre = base + r_SR"]
        PRE["Pre-projection HR<br/>B x 3 x 512 x 512"]
        DEGRADE["Differentiable sensor degradation<br/>HR to simulated LR"]
        ERROR["LR consistency error<br/>clean LR - degraded output"]
        PROJECT["Iterative sensor back-projection<br/>Default: 3 correction steps"]

        RESIDUAL --> ADD
        ADD --> PRE --> PROJECT --> OUTPUT

        DEG --> DEGRADE

    subgraph TRAINING["Training-Only Supervision"]

        HR -.-> PATCH
        OUTPUT -.-> PATCH
        HR -.-> WAVE
        OUTPUT -.-> WAVE
        HR -.-> LOSSES
        OUTPUT -.-> LOSSES
    end
